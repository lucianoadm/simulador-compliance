set.seed(1975) # Sua assinatura temporal mantida para reprodutibilidade
n_registros <- 5000

print("Iniciando a geração da massa de dados científica LexOS...")

# 1. Definição das Variáveis de Entrada Brutas
tipos_incidentes <- c("Vazamento_API", "Falha_Cookies", "Ausencia_Termos")
canais_origem    <- c("E-commerce", "Portal_Cliente", "App_Mobile")

id_incidente  <- 1000 + 1:n_registros
tipo_incidente <- sample(tipos_incidentes, n_registros, replace = TRUE, prob = c(0.4, 0.35, 0.25))
canal_origem   <- sample(canais_origem, n_registros, replace = TRUE)

# Simulação de variáveis numéricas contínuas independentes
volume_titulares <- round(runif(n_registros, min = 100, max = 50000))
severidade_reputacional <- round(runif(n_registros, min = 1, max = 10))

# 2. [AJUSTE Nº 5] Construção do ILR (Índice LexOS de Risco)
# Pesos metodológicos: Severidade (40%), Volume Normalizado (30%), Peso do Tipo (30%)
peso_tipo <- ifelse(tipo_incidente == "Vazamento_API", 10, 
                    ifelse(tipo_incidente == "Falha_Cookies", 6, 3))

# Normalização do volume de titulares para escala de 1 a 10
volume_norm <- (volume_titulares - 100) / (50000 - 100) * 9 + 1

# Cálculo matemático puro do ILR (Escala de 0 a 100)
ilr_puro <- (severidade_reputacional * 0.4 + volume_norm * 0.3 + peso_tipo * 0.3) * 10
ilr_final <- round(ilr_puro, 1)

# 3. Definição da Classe Alvo Teórica (Sem entregar a resposta mastigada)
resultado_final <- ifelse(ilr_final >= 72, "Alto_Risco",
                          ifelse(ilr_final >= 45, "Medio_Risco", "Baixo_Risco"))

# 4. [AJUSTE Nº 2] Introdução de Ruído Estatístico Real (Aproximadamente 7%)
# Força o Weka a aprender com imperfeições do mundo real (ex: ANPD/Procon)
for(i in 1:n_registros) {
  sorteio_ruido <- runif(1)
  
  if(sorteio_ruido < 0.07) { # 7% de anomalias na matriz jurídica
    classes_possiveis <- c("Alto_Risco", "Medio_Risco", "Baixo_Risco")
    # Substitui por uma classe aleatória diferente da real para simular "ruído"
    classes_erradas <- classes_possiveis[classes_possiveis != resultado_final[i]]
    resultado_final[i] <- sample(classes_erradas, 1)
  }
}

# 5. [AJUSTE Nº 1] Consolidação do Data Frame Científico 
# Removemos a probabilidade e a regra mastigada. O Weka terá que descobrir as fronteiras do ILR!
dados_lexos_cientifico <- data.frame(
  id = id_incidente,
  tipo_incidente = tipo_incidente,
  canal_origem = canal_origem,
  volume_titulares = volume_titulares,
  severidade_reputacional = severidade_reputacional,
  ilr = ilr_final,
  resultado_final = as.factor(resultado_final)
)

# 6. Salvando o Arquivo para o Experimento
write.csv(dados_lexos_cientifico, "dados_lexos_cientifico.csv", row.names = FALSE)
print("Massa de dados científica gerada com sucesso! Arquivo: dados_lexos_cientifico.csv")
